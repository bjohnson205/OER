\begin{align*} 4 \amp= \frac{2\pi}{|B|} \\ \\ |B| \cdot 4 \amp= 2\pi \\ \\ |B| \amp= \frac{2\pi}{4} \\ \\ |B| \amp= \frac{\pi}{2} \end{align*}

\begin{equation*} B = \frac{\pi}{2} \end{equation*}

\begin{equation*} k=3 \end{equation*}
