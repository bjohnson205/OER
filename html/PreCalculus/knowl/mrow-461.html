\begin{align*} \frac{\cos^2(\theta)+\sin^2(\theta)}{\sin^2(\theta)} \amp = \frac{1}{\sin^2(\theta)}\\ \\ \frac{\cos^2(\theta)}{\sin^2(\theta)} + \frac{\sin^2(\theta)}{\sin^2(\theta)} \amp = \frac{1}{\sin^2(\theta)}\\ \\ \frac{\cos^2(\theta)}{\sin^2(\theta)} + 1 \amp = \frac{1}{\sin^2(\theta)}\\ \\ \cot^2(\theta) + 1 \amp = \frac{1}{\sin^2(\theta)}\\ \\ \cot^2(\theta) + 1 \amp = \csc^2(\theta). \end{align*}
