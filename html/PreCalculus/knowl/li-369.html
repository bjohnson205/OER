\begin{equation*} f(\alert{2}) = \alert{2}^3 + 2 = \blert{10}. \end{equation*}

\begin{equation*} f^{-1}(\blert{10}) = \sqrt[3]{\blert{10} - 2} = \sqrt[3]{8}= \alert{2}. \end{equation*}
