\begin{align*} \int_e^{\infty} \frac{\ln(x)}{x} \, dx &= \lim_{b \to \infty} \int_e^{b} \frac{\ln(x)}{x} \, dx\\ &= \lim_{b \to \infty} \int_1^{\ln(b)} u \, du\\ &= \lim_{b \to \infty} \left( \frac{u^2}{2} \right)\biggm|_1^{\ln(b)}\\ &= \lim_{b \to \infty} \left( \frac{(\ln(b))^2}{2} - \frac{1}{2} \right)\\ &= \infty \end{align*}
