\begin{align*} \int_e^{\infty} \frac{1}{x(\ln(x))^p} \, dx &= \lim_{b \to \infty} \int_e^{b} \frac{1}{x(\ln(x))^p} \, dx\\ &= \lim_{b \to \infty} \int_1^{\ln(b)} \frac{1}{u^p} \, du\\ &= \lim_{b \to \infty} \int_1^{\ln(b)} u^{-p} \, du\\ &= \lim_{b \to \infty} \left( \frac{u^{-p+1}}{-p+1} \right)\biggm|_1^{\ln(b)}\\ &= \frac{1}{1-p} \lim_{b \to \infty} \left( (\ln(b))^{1-p} - 1 \right)\\ &= 1 \end{align*}
