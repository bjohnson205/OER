\begin{align*} \int_0^1 \frac{\ln(x)}{x} \, dx &= \lim_{a \to 0^+} \int_a^1 \frac{\ln(x)}{x} \, dx\\ &= \lim_{a \to 0^+} \int_{\ln(a)}^{0} u \, du\\ &= \lim_{a \to 0^+} \left( \frac{u^2}{2} \right)\biggm|_{\ln(a)}^{0}\\ &= \lim_{a \to 0^+} \left( -\frac{(\ln(a))^2}{2} \right)\\ &= -\infty \end{align*}
